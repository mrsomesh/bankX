CREATE TABLE `transactions` (
  `id` int(4) NOT NULL auto_increment,
  `account_number` varchar(20) NOT NULL default '',
  `amount` int(15) NOT NULL default '0',
  `details` varchar(50) NOT NULL default '',
  `time` varchar(30) NOT NULL default '',
  `day` varchar(30) NOT NULL default '',
  PRIMARY KEY  (`id`)
)


--------------------------------------------------------------------------------------------

CREATE TABLE `credentials` (
  `id` int(4) NOT NULL auto_increment,
  `account_number` varchar(20) NOT NULL default '',
  `unique_id` varchar(30) NOT NULL default '',
  `username` varchar(30) NOT NULL default '',
  `password` varchar(80) NOT NULL default '',
  `gender` varchar(10) NOT NULL default '',
  `email` varchar(80) NOT NULL default '',
  `mobile` varchar(20) NOT NULL default '',
  `state` varchar(80) NOT NULL default '',
  `pincode` varchar(20) NOT NULL default '',
  `authorization` varchar(8) NOT NULL default '',
  PRIMARY KEY  (`id`),
  UNIQUE KEY `account_number` (`account_number`),
  UNIQUE KEY `unique_id` (`unique_id`)
)


-------------------------------------------------------------------------------------------

CREATE TABLE `user_account` (
  `id` int(4) NOT NULL auto_increment,
  `account_number` varchar(20) NOT NULL default '',
  `unique_id` varchar(30) NOT NULL default '',
  `username` varchar(30) NOT NULL default '',
  `balance` int(15) NOT NULL default '0',
  `day` varchar(30) NOT NULL default '',
  PRIMARY KEY  (`id`),
  UNIQUE KEY `account_number` (`account_number`),
  UNIQUE KEY `unique_id` (`unique_id`)
)
